    ```csharp
    namespace Domain.Products;

    public class ProductDownload
    {
        public Guid Id { get; private set; }
        public Guid ProductId { get; private set; }
        public string FileName { get; private set; } = string.Empty;
        public string FilePath { get; private set; } = string.Empty; // Path in storage
        public string ContentType { get; private set; } = "application/octet-stream";
        public long FileSizeBytes { get; private set; }
        public DateTime CreatedAtUtc { get; private set; }

        // Factory
        public static ProductDownload Create(Guid productId, string fileName, string filePath, long fileSize)
        {
            return new ProductDownload
            {
                Id = Guid.NewGuid(),
                ProductId = productId,
                FileName = fileName,
                FilePath = filePath,
                FileSizeBytes = fileSize,
                ContentType = GetContentType(fileName),
                CreatedAtUtc = DateTime.UtcNow
            };
        }

        private static string GetContentType(string fileName)
        {
            var ext = Path.GetExtension(fileName).ToLowerInvariant();
            return ext switch
            {
                ".zip" => "application/zip",
                ".txt" => "text/plain",
                ".lua" => "text/plain",
                ".xml" => "application/xml",
                _ => "application/octet-stream"
            };
        }
    }
    ```

    ```csharp
    namespace Domain.Products;

    public class DownloadAccessPolicy
    {
        public Guid Id { get; private set; }
        public string PolicyName { get; private set; } = string.Empty; // e.g., "FreeScriptDownloads"
        public Guid? ProductCategoryId { get; private set; } // Null = all categories
        public bool AllowFreeDownload { get; private set; }
        public Guid[] AllowedRoles { get; private set; } = []; // Empty = all roles
        public DateTime CreatedAtUtc { get; private set; }

        public static DownloadAccessPolicy CreateFreePolicy(string name, Guid? categoryId, Guid[] allowedRoles)
        {
            return new DownloadAccessPolicy
            {
                Id = Guid.NewGuid(),
                PolicyName = name,
                ProductCategoryId = categoryId,
                AllowFreeDownload = true,
                AllowedRoles = allowedRoles,
                CreatedAtUtc = DateTime.UtcNow
            };
        }

        public bool AllowsDownload(Domain.Identity.UserRole userRole, bool hasPurchased)
        {
            // Paid downloads: must have purchased (entitlement check)
            if (!AllowFreeDownload)
                return hasPurchased;

            // Free downloads: check role allowlist
            if (AllowedRoles.Length == 0)
                return true; // All roles allowed

            return AllowedRoles.Contains(userRole);
        }
    }
    ```

    ```csharp
    using Application.Products.Contracts;
    using Microsoft.AspNetCore.Authorization;

    namespace API.Downloads;

    public static class DownloadEndpoints
    {
        public static void MapDownloadEndpoints(this IEndpointRouteBuilder app)
        {
            var group = app.MapGroup("/api/downloads")
                .WithTags("Downloads");

            // Generate signed download URL (requires purchase)
            group.MapPost("/generate-url", GenerateDownloadUrl)
                .RequireAuthorization();

            // Download file via signed URL (public endpoint with token)
            group.MapGet("/file/{token}", DownloadFile)
                .AllowAnonymous();
        }

        private static async Task<IResult> GenerateDownloadUrl(
            [FromBody] GenerateDownloadUrlRequest request,
            IDownloadEntitlementService entitlementService,
            Guid userId) // From auth
        {
            var result = await entitlementService.GenerateSignedUrlAsync(request.ProductId, userId);
            if (result == null)
                return Results.NotFound(new { error = "Download not available or not entitled" });

            return Results.Ok(result);
        }

        private static async Task<IResult> DownloadFile(
            string token,
            IFileService fileService)
        {
            // Validate signed URL token, check expiration
            // Stream file with proper Content-Type
            var (stream, metadata) = await fileService.GetFileAsync(token);
            return Results.File(stream, metadata.ContentType, metadata.FileName);
        }
    }

    public record GenerateDownloadUrlRequest(Guid ProductId);

    public interface IFileService
    {
        Task<(Stream Stream, FileMetadata Metadata)> GetFileAsync(string signedToken);
    }

    public record FileMetadata(string FileName, string ContentType, long Size);
    ```
