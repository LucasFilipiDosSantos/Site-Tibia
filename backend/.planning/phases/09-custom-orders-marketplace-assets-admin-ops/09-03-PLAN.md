    ```csharp
    using Application.Audit.Contracts;
    using Microsoft.AspNetCore.Authorization;

    namespace API.Admin;

    public static class AdminAuditEndpoints
    {
        public static void MapAdminAuditEndpoints(this IEndpointRouteBuilder app)
        {
            var group = app.MapGroup("/api/admin/audit")
                .WithTags("Admin - Audit")
                .RequireAuthorization("AdminOnly");

            group.MapGet("/logs", GetAuditLogs);
            group.MapGet("/logs/{id}", GetAuditLogById);
        }

        private static async Task<IResult> GetAuditLogs(
            [FromQuery] DateTime? from,
            [FromQuery] DateTime? to,
            [FromQuery] string? action,
            [FromQuery] string? entityType,
            [FromQuery] Guid? actorId,
            [FromQuery] Guid? entityId,
            [FromQuery] int page = 1,
            [FromQuery] int pageSize = 20,
            IAuditLogService? auditService = null)
        {
            var query = new AuditLogQuery(from, to, action, entityType, actorId, entityId, page, pageSize);
            var logs = await auditService!.QueryAsync(query);
            return Results.Ok(logs);
        }

        private static async Task<IResult> GetAuditLogById(
            Guid id,
            IAuditLogService? auditService)
        {
            var log = await auditService!.QueryAsync(new AuditLogQuery());
            var found = log.FirstOrDefault(l => l.Id == id);
            return found != null ? Results.Ok(found) : Results.NotFound();
        }
    }
    ```

    ```csharp
                .WithTags("Admin - Webhooks")
                .RequireAuthorization("AdminOnly");

            group.MapGet("/logs", GetWebhookLogs);
            group.MapGet("/logs/{id}", GetWebhookLogById);
        }

        private static async Task<IResult> GetWebhookLogs(
            [FromQuery] DateTime? from,
            [FromQuery] DateTime? to,
            [FromQuery] string? status,
            [FromQuery] string? paymentId,
            [FromQuery] int page = 1,
            [FromQuery] int pageSize = 20,
            IPaymentWebhookLogRepository? repo = null)
        {
            // Query PaymentWebhookLogRepository from Phase 6
            var logs = await repo!.QueryAsync(new WebhookLogQuery(from, to, status, paymentId, page, pageSize));
            return Results.Ok(logs);
        }

        private static async Task<IResult> GetWebhookLogById(
            Guid id,
            IPaymentWebhookLogRepository? repo)
        {
            var log = await repo!.GetByIdAsync(id);
            return log != null ? Results.Ok(log) : Results.NotFound();
        }
    }

    public record WebhookLogQuery(
        DateTime? From,
        DateTime? To,
        string? Status,
        string? PaymentId,
    ```
